# Criar um novo projeto Django no diretório atual
django-admin startproject nome_do_projeto .


# Criar um novo App dentro do projeto
python manage.py startapp nome_do_app


# Rodar o servidor de desenvolvimento
python manage.py runserver


# Coletar todos os arquivos estáticos (CSS, JS, imagens)
# dos apps e da pasta STATICFILES_DIRS para a pasta STATIC_ROOT
# Usado principalmente em ambiente de produção
python manage.py collectstatic .
